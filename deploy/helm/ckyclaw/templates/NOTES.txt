   - Backend (FastAPI) — {{ .Values.backend.replicaCount }} replica(s)
   - Frontend (React/Nginx) — {{ .Values.frontend.replicaCount }} replica(s)
   {{- if .Values.postgresql.enabled }}

   {{- else }}
   - External PostgreSQL: {{ .Values.externalDatabase.host }}:{{ .Values.externalDatabase.port }}
   {{- end }}
   {{- if .Values.redis.enabled }}
   - Redis 7 (built-in)
   {{- else }}

   {{- end }}

   {{- if .Values.ingress.enabled }}
   {{- range .Values.ingress.hosts }}
   URL: http{{ if $.Values.ingress.tls }}s{{ end }}://{{ .host }}
